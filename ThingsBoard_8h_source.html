234 bool Send_Json(char const * const topic, JsonDocument const & source, size_t const & jsonSize) {
235 // Check if allocating needed memory failed when trying to create the JsonDocument,
236 // if it did the isNull() method will return true. See https://arduinojson.org/v6/api/jsonvariant/isnull/ for more information
237 if (source.isNull()) {
238 Logger::println(UNABLE_TO_ALLOCATE_JSON);
239 return false;
240 }
241 // Check if inserting any of the internal values failed because the JsonDocument was too small,
242 // if it did the overflowed() method will return true. See https://arduinojson.org/v6/api/jsondocument/overflowed/ for more information
243 if (source.overflowed()) {
244 Logger::println(JSON_SIZE_TO_SMALL);
245 return false;
246 }
247#if !THINGSBOARD_ENABLE_DYNAMIC
248 size_t const amount = source.size();
249 if (MaxFieldsAmount < amount) {
250 Logger::printfln(TOO_MANY_JSON_FIELDS, amount, MaxFieldsAmount);
251 return false;
252 }
253#endif // !THINGSBOARD_ENABLE_DYNAMIC
254 bool result = false;
255
256#if THINGSBOARD_ENABLE_STREAM_UTILS
257 // Check if the size of the given message would be too big for the actual client,
258 // if it is utilize the serialize json work around, so that the internal client buffer can be circumvented
259 if (m_client.get_buffer_size() < jsonSize) {
260#if THINGSBOARD_ENABLE_DEBUG
261 Logger::printfln(SEND_MESSAGE, topic, SEND_SERIALIZED);
262#endif // THINGSBOARD_ENABLE_DEBUG
263 result = Serialize_Json(topic, source, jsonSize - 1);
264 }
265 // Check if the remaining stack size of the current task would overflow the stack,
266 // if it would allocate the memory on the heap instead to ensure no stack overflow occurs
267 else
268#endif // THINGSBOARD_ENABLE_STREAM_UTILS
269 if (getMaximumStackSize() < jsonSize) {
270 char* json = new char[jsonSize]();
271 if (serializeJson(source, json, jsonSize) < jsonSize - 1) {
272 Logger::println(UNABLE_TO_SERIALIZE_JSON);
273 }
274 else {
275 result = Send_Json_String(topic, json);
276 }
277 // Ensure to actually delete the memory placed onto the heap, to make sure we do not create a memory leak
278 // and set the pointer to null so we do not have a dangling reference.
279 delete[] json;
280 json = nullptr;
281 }
282 else {
283 char json[jsonSize] = {};
284 if (serializeJson(source, json, jsonSize) < jsonSize - 1) {
285 Logger::println(UNABLE_TO_SERIALIZE_JSON);
286 return result;
287 }
288 result = Send_Json_String(topic, json);
289 }
290
291 return result;
292 }

298 bool Send_Json_String(char const * const topic, char const * const json) {
299 if (json == nullptr) {
300 return false;
301 }
302
303 uint16_t currentBufferSize = m_client.get_buffer_size();
304 size_t const jsonSize = strlen(json);
305
306 if (currentBufferSize < jsonSize) {
307 Logger::printfln(INVALID_BUFFER_SIZE, currentBufferSize, jsonSize);
308 return false;
309 }

313#endif // THINGSBOARD_ENABLE_DEBUG
314 return m_client.publish(topic, reinterpret_cast<uint8_t const *>(json), jsonSize);
315 }

538 bool sendKeyValue(char const * const key, T const & value, bool telemetry = true) {
539 const Telemetry t(key, value);
540 if (t.IsEmpty()) {
541 return false;
542 }
543
544 StaticJsonDocument<JSON_OBJECT_SIZE(1)> jsonBuffer;
545 if (!t.SerializeKeyValue(jsonBuffer)) {
546 Logger::println(UNABLE_TO_SERIALIZE);
547 return false;
548 }
549 return telemetry ? sendTelemetryJson(jsonBuffer, Helper::Measure_Json(jsonBuffer)) : sendAttributeJson(jsonBuffer, Helper::Measure_Json(jsonBuffer));

560 template<typename InputIterator>
561 bool sendDataArray(InputIterator const & first, InputIterator const & last, bool telemetry) {
562#if THINGSBOARD_ENABLE_DYNAMIC
563 // String are char const * and therefore stored as a pointer --> zero copy, meaning the size for the strings is 0 bytes,
564 // Data structure size depends on the amount of key value pairs passed.
565 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
566 size_t const size = Helper::distance(first, last);
567 TBJsonDocument jsonBuffer(JSON_OBJECT_SIZE(size));
568#else
569 StaticJsonDocument<JSON_OBJECT_SIZE(MaxFieldsAmount)> jsonBuffer;
570#endif // THINGSBOARD_ENABLE_DYNAMIC
571
572 for (auto it = first; it != last; ++it) {
573 auto const & data = *it;
574 if (!data.SerializeKeyValue(jsonBuffer)) {
575 Logger::println(UNABLE_TO_SERIALIZE);
576 return false;
577 }
578 }
579 return telemetry ? sendTelemetryJson(jsonBuffer, Helper::Measure_Json(jsonBuffer)) : sendAttributeJson(jsonBuffer, Helper::Measure_Json(jsonBuffer));
580 }

586 void onMQTTMessage(char * topic, uint8_t * payload, unsigned int length) {
587#if THINGSBOARD_ENABLE_DEBUG
588 Logger::printfln(RECEIVE_MESSAGE, topic);
589#endif // THINGSBOARD_ENABLE_DEBUG
590
591 for (auto & api : m_api_implementations) {
592 if (api == nullptr || api->Get_Process_Type() != API_Process_Type::RAW || strncmp(topic, api->Get_Response_Topic_String(), strlen(topic)) != 0) {
593 continue;
594 }
595 // Check if the remaining stack size of the current task would overflow the stack,
596 // if it would allocate the memory on the heap instead to ensure no stack overflow occurs.
597 if (getMaximumStackSize() < length) {
598 uint8_t* binary = new uint8_t[length]();
599 (void)memcpy(binary, payload, length);
600 api->Process_Response(topic, binary, length);
601 // Ensure to actually delete the memory placed onto the heap, to make sure we do not create a memory leak
602 // and set the pointer to null so we do not have a dangling reference.
603 delete[] binary;
604 binary = nullptr;
605 }
606 else {
607 uint8_t binary[length] = {};
608 (void)memcpy(binary, payload, length);
609 api->Process_Response(topic, binary, length);
610 }
611 return;
612 }
613
614#if THINGSBOARD_ENABLE_DYNAMIC
615 // Buffer that we deserialize is writeable and not read only --> zero copy, meaning the size for the data is 0 bytes,
616 // Data structure size depends on the amount of key value pairs received.
617 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
618 TBJsonDocument jsonBuffer(JSON_OBJECT_SIZE(Helper::getOccurences(reinterpret_cast<char *>(payload), ':')));
619#else
620 StaticJsonDocument<JSON_OBJECT_SIZE(MaxFieldsAmount)> jsonBuffer;
621#endif // THINGSBOARD_ENABLE_DYNAMIC
622
623 // The deserializeJson method we use, can use the zero copy mode because a writeable input was passed,
624 // if that were not the case the needed allocated memory would drastically increase, because the keys would need to be copied as well.
625 // See https://arduinojson.org/v6/doc/deserialization/ for more info on ArduinoJson deserialization
626 DeserializationError const error = deserializeJson(jsonBuffer, payload, length);
627 if (error) {
628 Logger::printfln(UNABLE_TO_DE_SERIALIZE_JSON, error.c_str());
629 return;
630 }
631 // .as() is used instead of .to(), because it is meant to cast the JsonDocument to the given type,
632 // but it does not change the actual content of the JsonDocument, we don't want that because it contains content
633 // and .to() would result in the data being cleared ()"null"), instead .as() which allows accessing the data over a JsonObjectConst instead
634 JsonObjectConst data = jsonBuffer.template as<JsonObjectConst>();
635
636 for (auto & api : m_api_implementations) {
637 if (api == nullptr || api->Get_Process_Type() != API_Process_Type::JSON || strncmp(topic, api->Get_Response_Topic_String(), strlen(topic)) != 0) {
638 continue;
639 }
640 api->Process_Json_Response(topic, data);
641 }
642 }
