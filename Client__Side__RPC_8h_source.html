46 bool RPC_Request(RPC_Request_Callback const & callback) {
47 char const * methodName = callback.Get_Name();
48
49 if (Helper::stringIsNullorEmpty(methodName)) {
50 Logger::println(CLIENT_RPC_METHOD_NULL);
51 return false;
52 }
53 RPC_Request_Callback * registeredCallback = nullptr;
54 if (!RPC_Request_Subscribe(callback, registeredCallback)) {
55 return false;
56 }
57 else if (registeredCallback == nullptr) {
58 return false;
59 }
60
61 JsonArray const * const parameters = callback.Get_Parameters();
62
63#if THINGSBOARD_ENABLE_DYNAMIC
64 // String are const char* and therefore stored as a pointer --> zero copy, meaning the size for the strings is 0 bytes,
65 // Data structure size depends on the amount of key value pairs passed + the default methodName and params key needed for the request.
66 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
67 TBJsonDocument requestBuffer(JSON_OBJECT_SIZE(parameters != nullptr ? parameters->size() + 2U : 2U));
68#else
69 // Ensure to have enough size for the infinite amount of possible parameters that could be sent to the cloud
70 StaticJsonDocument<JSON_OBJECT_SIZE(MaxRequestRPC)> requestBuffer;
71#endif // THINGSBOARD_ENABLE_DYNAMIC
72
73 requestBuffer[RPC_METHOD_KEY] = methodName;
74
75 if (parameters != nullptr && !parameters->isNull()) {
76 requestBuffer[RPC_PARAMS_KEY] = *parameters;
77 }
78 else {
79 requestBuffer[RPC_PARAMS_KEY] = RPC_EMPTY_PARAMS_VALUE;
80 }
81
82#if !THINGSBOARD_ENABLE_DYNAMIC
83 if (requestBuffer.overflowed()) {
84 Logger::printfln(RPC_REQUEST_OVERFLOWED, MaxRequestRPC);
85 return false;
86 }
87#endif // !THINGSBOARD_ENABLE_DYNAMIC
88
89 m_request_id++;
90 registeredCallback->Set_Request_ID(m_request_id);
91 registeredCallback->Start_Timeout_Timer();
92
93 char topic[Helper::detectSize(RPC_SEND_REQUEST_TOPIC, m_request_id)] = {};
94 (void)snprintf(topic, sizeof(topic), RPC_SEND_REQUEST_TOPIC, m_request_id);
95
96 size_t const objectSize = Helper::Measure_Json(requestBuffer);
97 return m_send_callback.Call_Callback(topic, requestBuffer, objectSize);
98 }

116 void Process_Json_Response(char * const topic, JsonObjectConst & data) override {
117 size_t const request_id = Helper::parseRequestId(RPC_RESPONSE_TOPIC, topic);
118
119 for (auto it = m_rpc_request_callbacks.begin(); it != m_rpc_request_callbacks.end(); ++it) {
120 auto & rpc_request = *it;
121
122 if (rpc_request.Get_Request_ID() != request_id) {
123 continue;
124 }
125 rpc_request.Stop_Timeout_Timer();
126 rpc_request.Call_Callback(data);
127
128 // Delete callback because the changes have been requested and the callback is no longer needed
129 Helper::remove(m_rpc_request_callbacks, it);
130 break;
131 }
132
133 // Attempt to unsubscribe from the shared attribute request topic,
134 // if we are not waiting for any further responses with shared attributes from the server.
135 // Will be resubscribed if another request is sent anyway
136 if (m_rpc_request_callbacks.empty()) {
137 (void)RPC_Request_Unsubscribe();
138 }
139 }

148 bool RPC_Request_Subscribe(RPC_Request_Callback const & callback, RPC_Request_Callback * & registeredCallback) {
149#if !THINGSBOARD_ENABLE_DYNAMIC
150 if (m_rpc_request_callbacks.size() + 1 > m_rpc_request_callbacks.capacity()) {
151 Logger::printfln(MAX_SUBSCRIPTIONS_EXCEEDED, CLIENT_SIDE_RPC_SUBSCRIPTIONS);
152 return false;
153 }
154#endif // !THINGSBOARD_ENABLE_DYNAMIC
155 if (!m_subscribe_callback.Call_Callback(RPC_RESPONSE_SUBSCRIBE_TOPIC)) {
156 Logger::printfln(SUBSCRIBE_TOPIC_FAILED, RPC_RESPONSE_SUBSCRIBE_TOPIC);
157 return false;
158 }
159
160 // Push back given callback into our local vector
161 m_rpc_request_callbacks.push_back(callback);
162 registeredCallback = &m_rpc_request_callbacks.back();
163 return true;
164 }

169 bool RPC_Request_Unsubscribe() {
170 m_rpc_request_callbacks.clear();
171 return m_unsubscribe_callback.Call_Callback(RPC_RESPONSE_SUBSCRIBE_TOPIC);
172 }

174 // Vectors or array (depends on wheter if THINGSBOARD_ENABLE_DYNAMIC is set to 1 or 0), hold copy of the actual passed data, this is to ensure they stay valid,
175 // even if the user only temporarily created the object before the method was called.
176 // This can be done because all Callback methods mostly consists of pointers to actual object so copying them
177 // does not require a huge memory overhead and is acceptable especially in comparsion to possible problems that could
178 // arise if references were used and the end user does not take care to ensure the Callbacks live on for the entirety
179 // of its usage, which will lead to dangling references and undefined behaviour.
180 // Therefore copy-by-value has been choosen as for this specific use case it is more advantageous,
181 // especially because at most we copy internal vectors or array, that will only ever contain a few pointers
182#if THINGSBOARD_ENABLE_DYNAMIC
183 Vector<RPC_Request_Callback> m_rpc_request_callbacks; // Server side RPC callbacks vector
184#else
185 Array<RPC_Request_Callback, MaxSubscribtions> m_rpc_request_callbacks; // Server side RPC callbacks array
186#endif // THINGSBOARD_ENABLE_DYNAMIC
187 static size_t m_request_id; // Allows nearly 4.3 million requests before wrapping back to 0, static so we actually keep track of the current request id even if we use multiple instances
188};
