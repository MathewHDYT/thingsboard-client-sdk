118 void Process_Json_Response(char * const topic, JsonObjectConst & data) override {
119 char const * const methodName = data[RPC_METHOD_KEY];
120
121 if (methodName == nullptr) {
122 Logger::println(SERVER_RPC_METHOD_NULL);
123 return;
124 }
125
126 for (auto const & rpc : m_rpc_callbacks) {
127 char const * const subscribedMethodName = rpc.Get_Name();
128 if (Helper::stringIsNullorEmpty(subscribedMethodName)) {
129 Logger::println(SERVER_RPC_METHOD_NULL);
130 continue;
131 }
132 // Strncmp returns the ascii value difference of the ascii characters that are different,
133 // meaning 0 is the same string and less and more than 0 is the difference in ascci values between the 2 chararacters.
134 else if (strncmp(subscribedMethodName, methodName, strlen(subscribedMethodName)) != 0) {
135 continue;
136 }
137
138 // Do not inform client, if parameter field is missing for some reason
139 if (!data.containsKey(RPC_PARAMS_KEY)) {
140#if THINGSBOARD_ENABLE_DEBUG
141 Logger::println(NO_RPC_PARAMS_PASSED);
142#endif // THINGSBOARD_ENABLE_DEBUG
143 }
144
145#if THINGSBOARD_ENABLE_DEBUG
146 Logger::printfln(CALLING_RPC_CB, methodName);
147#endif // THINGSBOARD_ENABLE_DEBUG
148
149 JsonVariantConst const param = data[RPC_PARAMS_KEY];
150#if THINGSBOARD_ENABLE_DYNAMIC
151 size_t const & rpc_response_size = rpc.Get_Response_Size();
152 // String are char const * and therefore stored as a pointer --> zero copy, meaning the size for the strings is 0 bytes,
153 // Data structure size depends on the amount of key value pairs passed.
154 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
155 TBJsonDocument jsonBuffer(rpc_response_size);
156#else
157 size_t constexpr rpc_response_size = MaxRPC;
158 StaticJsonDocument<JSON_OBJECT_SIZE(MaxRPC)> jsonBuffer;
159#endif // THINGSBOARD_ENABLE_DYNAMIC
160 rpc.Call_Callback(param, jsonBuffer);
161
162 if (jsonBuffer.isNull()) {
163#if THINGSBOARD_ENABLE_DEBUG
164 Logger::println(RPC_RESPONSE_NULL);
165#endif // THINGSBOARD_ENABLE_DEBUG
166 break;
167 }
168 else if (jsonBuffer.overflowed()) {
169 Logger::printfln(RPC_RESPONSE_OVERFLOWED, rpc_response_size);
170 break;
171 }
172
173 size_t const request_id = Helper::parseRequestId(RPC_REQUEST_TOPIC, topic);
174 char responseTopic[Helper::detectSize(RPC_SEND_RESPONSE_TOPIC, request_id)] = {};
175 (void)snprintf(responseTopic, sizeof(responseTopic), RPC_SEND_RESPONSE_TOPIC, request_id);
176
177 size_t const jsonSize = Helper::Measure_Json(jsonBuffer);
178 (void)m_send_callback.Call_Callback(responseTopic, jsonBuffer, jsonSize);
179 break;
180 }
181 }

184 // Vectors or array (depends on wheter if THINGSBOARD_ENABLE_DYNAMIC is set to 1 or 0), hold copy of the actual passed data, this is to ensure they stay valid,
185 // even if the user only temporarily created the object before the method was called.
186 // This can be done because all Callback methods mostly consists of pointers to actual object so copying them
187 // does not require a huge memory overhead and is acceptable especially in comparsion to possible problems that could
188 // arise if references were used and the end user does not take care to ensure the Callbacks live on for the entirety
189 // of its usage, which will lead to dangling references and undefined behaviour.
190 // Therefore copy-by-value has been choosen as for this specific use case it is more advantageous,
191 // especially because at most we copy internal vectors or array, that will only ever contain a few pointers
192#if THINGSBOARD_ENABLE_DYNAMIC
