117 if (!data) {
118#if THINGSBOARD_ENABLE_DEBUG

120#endif // THINGSBOARD_ENABLE_DEBUG

123
124 if (data.containsKey(SHARED_RESPONSE_KEY)) {
125 data = data[SHARED_RESPONSE_KEY];
126 }
127
128 for (auto const & shared_attribute : m_shared_attribute_update_callbacks) {
129 if (shared_attribute.Get_Attributes().empty()) {
130#if THINGSBOARD_ENABLE_DEBUG
131 Logger::println(ATT_CB_NO_KEYS);
132#endif // THINGSBOARD_ENABLE_DEBUG
133 // No specifc keys were subscribed so we call the callback anyway, assumed to be subscribed to any update
134 shared_attribute.Call_Callback(data);
135 continue;
136 }
137
138 char const * requested_att = nullptr;
139
140 for (auto const & att : shared_attribute.Get_Attributes()) {
141 if (Helper::stringIsNullorEmpty(att)) {
142#if THINGSBOARD_ENABLE_DEBUG
143 Logger::println(SHARED_KEY_IS_NULL);
144#endif // THINGSBOARD_ENABLE_DEBUG
145 continue;
146 }
147 // Check if the request contained any of our requested keys and
148 // break early if the key was requested from this callback.
149 if (data.containsKey(att)) {
150 requested_att = att;
151 break;
152 }
153 }
154
155 // Check if this callback did not request any keys that were in this response,
156 // if there were not we simply continue with the next subscribed callback.
157 if (requested_att == nullptr) {
158#if THINGSBOARD_ENABLE_DEBUG
159 Logger::println(ATT_NO_CHANGE);
160#endif // THINGSBOARD_ENABLE_DEBUG
161 continue;
162 }
163
164#if THINGSBOARD_ENABLE_DEBUG
165 Logger::printfln(CALLING_ATT_CB, requested_att);
166#endif // THINGSBOARD_ENABLE_DEBUG
167 shared_attribute.Call_Callback(data);
168 }

172 // Vectors or array (depends on wheter if THINGSBOARD_ENABLE_DYNAMIC is set to 1 or 0), hold copy of the actual passed data, this is to ensure they stay valid,
173 // even if the user only temporarily created the object before the method was called.
174 // This can be done because all Callback methods mostly consists of pointers to actual object so copying them
175 // does not require a huge memory overhead and is acceptable especially in comparsion to possible problems that could
176 // arise if references were used and the end user does not take care to ensure the Callbacks live on for the entirety
177 // of its usage, which will lead to dangling references and undefined behaviour.
178 // Therefore copy-by-value has been choosen as for this specific use case it is more advantageous,
179 // especially because at most we copy internal vectors or array, that will only ever contain a few pointers
