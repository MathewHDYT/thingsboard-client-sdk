238 bool Firmware_OTA_Unsubscribe() {
239 // Buffer size has been set to another value before the update,
240 // to allow to receive ota chunck packets that might be much bigger than the normal
241 // buffer size would allow, therefore we return to the previous value to decrease overall memory usage
242 if (m_change_buffer_size) {
243 (void)m_set_buffer_size.Call_Callback(m_previous_buffer_size);
244 }
245 // Reset now not needed private member variables
246 m_fw_callback = OTA_Update_Callback();
247 // Unsubscribe from the topic

279 void Firmware_Shared_Attribute_Received(JsonObjectConst const & data) {
280 // Check if firmware is available for our device
281 if (!data.containsKey(FW_VER_KEY) || !data.containsKey(FW_TITLE_KEY) || !data.containsKey(FW_CHKS_KEY) || !data.containsKey(FW_CHKS_ALGO_KEY) || !data.containsKey(FW_SIZE_KEY)) {
282 Logger::println(NO_FW);
283 Firmware_Send_State(FW_STATE_FAILED, NO_FW);
284 return;
285 }
286
287 char const * const fw_title = data[FW_TITLE_KEY];
288 char const * const fw_version = data[FW_VER_KEY];
289 char const * const fw_checksum = data[FW_CHKS_KEY];
290 char const * const fw_algorithm = data[FW_CHKS_ALGO_KEY];
291 size_t const fw_size = data[FW_SIZE_KEY];
292
293 char const * const curr_fw_title = m_fw_callback.Get_Firmware_Title();
294 char const * const curr_fw_version = m_fw_callback.Get_Firmware_Version();
295
296 if (fw_title == nullptr || fw_version == nullptr || curr_fw_title == nullptr || curr_fw_version == nullptr || fw_algorithm == nullptr || fw_checksum == nullptr) {
297 Logger::println(EMPTY_FW);
298 Firmware_Send_State(FW_STATE_FAILED, EMPTY_FW);
299 return;
300 }
301 // If firmware version and title is the same, we do not initiate an update, because we expect the type of binary to be the same one we are currently using and therefore updating would be useless
302 else if (strncmp(curr_fw_title, fw_title, strlen(curr_fw_title)) == 0 && strncmp(curr_fw_version, fw_version, strlen(curr_fw_version)) == 0) {
303 char message[JSON_STRING_SIZE(strlen(FW_UP_TO_DATE)) + JSON_STRING_SIZE(strlen(curr_fw_version))] = {};
304 (void)snprintf(message, sizeof(message), FW_UP_TO_DATE, curr_fw_version);
305 Logger::println(message);
306 Firmware_Send_State(FW_STATE_FAILED, message);
307 return;
308 }
309 // If firmware title is not the same, we do not initiate an update, because we expect the binary to be for another type of device and downloading it on this device could possibly cause hardware issues
310 else if (strncmp(curr_fw_title, fw_title, strlen(curr_fw_title)) != 0) {
311 char message[JSON_STRING_SIZE(strlen(FW_NOT_FOR_US)) + JSON_STRING_SIZE(strlen(curr_fw_title)) + JSON_STRING_SIZE(strlen(fw_title))] = {};
312 (void)snprintf(message, sizeof(message), FW_NOT_FOR_US, curr_fw_title, fw_title);
313 Logger::println(message);
314 Firmware_Send_State(FW_STATE_FAILED, message);
315 return;
316 }
317
318 mbedtls_md_type_t fw_checksum_algorithm = mbedtls_md_type_t{};
319
320 if (strncmp(CHECKSUM_AGORITM_MD5, fw_algorithm, strlen(CHECKSUM_AGORITM_MD5)) == 0) {
321 fw_checksum_algorithm = mbedtls_md_type_t::MBEDTLS_MD_MD5;
322 }
323 else if (strncmp(CHECKSUM_AGORITM_SHA256, fw_algorithm, strlen(CHECKSUM_AGORITM_SHA256)) == 0) {
324 fw_checksum_algorithm = mbedtls_md_type_t::MBEDTLS_MD_SHA256;
325 }
326 else if (strncmp(CHECKSUM_AGORITM_SHA384, fw_algorithm, strlen(CHECKSUM_AGORITM_SHA384)) == 0) {
327 fw_checksum_algorithm = mbedtls_md_type_t::MBEDTLS_MD_SHA384;
328 }
329 else if (strncmp(CHECKSUM_AGORITM_SHA512, fw_algorithm, strlen(CHECKSUM_AGORITM_SHA512)) == 0) {
330 fw_checksum_algorithm = mbedtls_md_type_t::MBEDTLS_MD_SHA512;
331 }
332 else {
333 char message[JSON_STRING_SIZE(strlen(FW_CHKS_ALGO_NOT_SUPPORTED)) + JSON_STRING_SIZE(strlen(fw_algorithm))] = {};
334 (void)snprintf(message, sizeof(message), FW_CHKS_ALGO_NOT_SUPPORTED, fw_algorithm);
335 Logger::println(message);
336 Firmware_Send_State(FW_STATE_FAILED, message);
337 return;
338 }
339
340 if (!Firmware_OTA_Subscribe()) {
341 return;
342 }
343
344#if THINGSBOARD_ENABLE_DEBUG
345 Logger::println(PAGE_BREAK);
346 Logger::println(NEW_FW);
347 char firmware[JSON_STRING_SIZE(strlen(FROM_TOO)) + JSON_STRING_SIZE(strlen(curr_fw_version)) + JSON_STRING_SIZE(strlen(fw_version))] = {};
348 (void)snprintf(firmware, sizeof(firmware), FROM_TOO, curr_fw_version, fw_version);
349 Logger::println(firmware);
350 Logger::println(DOWNLOADING_FW);
351#endif // THINGSBOARD_ENABLE_DEBUG
352
353 // Calculate the number of chuncks we need to request,
354 // in order to download the complete firmware binary
355 const uint16_t& chunk_size = m_fw_callback.Get_Chunk_Size();
356
357 // Get the previous buffer size and cache it so the previous settings can be restored.
358 m_previous_buffer_size = m_get_size_callback.Call_Callback();
359 m_change_buffer_size = m_previous_buffer_size < (chunk_size + 50U);
360
361 // Increase size of receive buffer
362 if (m_change_buffer_size && !m_set_buffer_size.Call_Callback(chunk_size + 50U)) {
363 Logger::println(NOT_ENOUGH_RAM);
364 Firmware_Send_State(FW_STATE_FAILED, NOT_ENOUGH_RAM);
365 return;
366 }
367
368 m_ota.Start_Firmware_Update(m_fw_callback, fw_size, fw_checksum, fw_checksum_algorithm);
369 }
