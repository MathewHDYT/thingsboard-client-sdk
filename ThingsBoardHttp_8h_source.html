81 bool Send_Json(char const * const topic, JsonDocument const & source, size_t const & jsonSize) {
82 // Check if allocating needed memory failed when trying to create the JsonDocument,
83 // if it did the isNull() method will return true. See https://arduinojson.org/v6/api/jsonvariant/isnull/ for more information
84 if (source.isNull()) {
85 Logger::println(UNABLE_TO_ALLOCATE_JSON);
86 return false;
87 }
88 // Check if inserting any of the internal values failed because the JsonDocument was too small,
89 // if it did the overflowed() method will return true. See https://arduinojson.org/v6/api/jsondocument/overflowed/ for more information
90 if (source.overflowed()) {
91 Logger::println(JSON_SIZE_TO_SMALL);
92 return false;
93 }
94#if !THINGSBOARD_ENABLE_DYNAMIC
95 size_t const amount = source.size();
96 if (MaxFieldsAmount < amount) {
97 Logger::printfln(TOO_MANY_JSON_FIELDS, amount, MaxFieldsAmount);
98 return false;
99 }
100#endif // !THINGSBOARD_ENABLE_DYNAMIC
101 bool result = false;
102 if (getMaximumStackSize() < jsonSize) {
103 char * json = new char[jsonSize]();
104 if (serializeJson(source, json, jsonSize) < jsonSize - 1) {
105 Logger::println(UNABLE_TO_SERIALIZE_JSON);
106 }
107 else {
108 result = Send_Json_String(topic, json);
109 }
110 // Ensure to actually delete the memory placed onto the heap, to make sure we do not create a memory leak
111 // and set the pointer to null so we do not have a dangling reference.
112 delete[] json;
113 json = nullptr;
114 }
115 else {
116 char json[jsonSize] = {};
117 if (serializeJson(source, json, jsonSize) < jsonSize - 1) {
118 Logger::println(UNABLE_TO_SERIALIZE_JSON);
119 return result;
120 }
121 result = Send_Json_String(topic, json);
122 }
123 return result;
124 }

130 bool Send_Json_String(char const * const topic, char const * const json) {
131 if (json == nullptr || m_token == nullptr) {
132 return false;
133 }
134
135 char path[Helper::detectSize(topic, m_token)] = {};
136 (void)snprintf(path, sizeof(path), topic, m_token);
137 return postMessage(path, json);
138 }

272 bool postMessage(char const * const path, char const * const json) {
273 bool success = m_client.post(path, HTTP_POST_PATH, json) == 0;
274 int const status = m_client.get_response_status_code();
275
276 if (!success || status < HTTP_RESPONSE_SUCCESS_RANGE_START || status > HTTP_RESPONSE_SUCCESS_RANGE_END) {
277 Logger::printfln(HTTP_FAILED, POST, status);
278 success = false;
279 }
280
281 clearConnection();
282 return success;
283 }

318 template<typename InputIterator>
319 bool sendDataArray(InputIterator const & first, InputIterator const & last, bool telemetry) {
320#if THINGSBOARD_ENABLE_DYNAMIC
321 // String are const char* and therefore stored as a pointer --> zero copy, meaning the size for the strings is 0 bytes,
322 // Data structure size depends on the amount of key value pairs passed.
323 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
324 size_t const size = Helper::distance(first, last);
325 TBJsonDocument jsonBuffer(JSON_OBJECT_SIZE(size));
326#else
327 StaticJsonDocument<JSON_OBJECT_SIZE(MaxFieldsAmount)> jsonBuffer;
328#endif // !THINGSBOARD_ENABLE_DYNAMIC
329
330 for (auto it = first; it != last; ++it) {
331 auto const & data = *it;
332 if (!data.SerializeKeyValue(jsonBuffer)) {
333 Logger::println(UNABLE_TO_SERIALIZE);
334 return false;
335 }
336 }
337 return telemetry ? sendTelemetryJson(jsonBuffer, Helper::Measure_Json(jsonBuffer)) : sendAttributeJson(jsonBuffer, Helper::Measure_Json(jsonBuffer));
338 }
339
346 template<typename T>
347 bool sendKeyValue(char const * const key, T value, bool telemetry = true) {
348 Telemetry const t(key, value);
349 if (t.IsEmpty()) {
350 // Message is ignored and not sent at all.
351 return false;
352 }
353
354 StaticJsonDocument<JSON_OBJECT_SIZE(1)> jsonBuffer;
355 if (!t.SerializeKeyValue(jsonBuffer)) {
356 Logger::printfln(UNABLE_TO_SERIALIZE);
357 return false;
358 }
359 return telemetry ? sendTelemetryJson(jsonBuffer, Helper::Measure_Json(jsonBuffer)) : sendAttributeJson(jsonBuffer, Helper::Measure_Json(jsonBuffer));
360 }
