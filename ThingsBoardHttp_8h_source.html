81 template <typename TSource>
82 bool Send_Json(char const * const topic, TSource const & source, size_t const & jsonSize) {
83 // Check if allocating needed memory failed when trying to create the JsonObject,
84 // if it did the isNull() method will return true. See https://arduinojson.org/v6/api/jsonvariant/isnull/ for more information
85 if (source.isNull()) {
86 Logger::println(UNABLE_TO_ALLOCATE_JSON);
87 return false;
88 }
89#if !THINGSBOARD_ENABLE_DYNAMIC
90 size_t const amount = source.size();
91 if (MaxFieldsAmount < amount) {
92 Logger::printfln(TOO_MANY_JSON_FIELDS, amount, MaxFieldsAmount);
93 return false;
94 }
95#endif // !THINGSBOARD_ENABLE_DYNAMIC
96 bool result = false;
97 if (getMaximumStackSize() < jsonSize) {
98 char * json = new char[jsonSize]();
99 if (serializeJson(source, json, jsonSize) < jsonSize - 1) {
100 Logger::println(UNABLE_TO_SERIALIZE_JSON);
101 }
102 else {
103 result = Send_Json_String(topic, json);
104 }
105 // Ensure to actually delete the memory placed onto the heap, to make sure we do not create a memory leak
106 // and set the pointer to null so we do not have a dangling reference.
107 delete[] json;
108 json = nullptr;
109 }
110 else {
111 char json[jsonSize] = {};
112 if (serializeJson(source, json, jsonSize) < jsonSize - 1) {
113 Logger::println(UNABLE_TO_SERIALIZE_JSON);
114 return result;
115 }
116 result = Send_Json_String(topic, json);
117 }
118 return result;
119 }

125 bool Send_Json_String(char const * const topic, char const * const json) {
126 if (json == nullptr || m_token == nullptr) {
127 return false;
128 }
129
130 char path[Helper::detectSize(topic, m_token)] = {};
131 (void)snprintf(path, sizeof(path), topic, m_token);
132 return postMessage(path, json);
133 }

269 bool postMessage(char const * const path, char const * const json) {
270 bool success = m_client.post(path, HTTP_POST_PATH, json) == 0;
271 int const status = m_client.get_response_status_code();
272
273 if (!success || status < HTTP_RESPONSE_SUCCESS_RANGE_START || status > HTTP_RESPONSE_SUCCESS_RANGE_END) {
274 Logger::printfln(HTTP_FAILED, POST, status);
275 success = false;
276 }
277
278 clearConnection();
279 return success;
280 }

316 template<typename InputIterator>
317 bool sendDataArray(InputIterator const & first, InputIterator const & last, bool telemetry) {
318#if THINGSBOARD_ENABLE_DYNAMIC
319 // String are const char* and therefore stored as a pointer --> zero copy, meaning the size for the strings is 0 bytes,
320 // Data structure size depends on the amount of key value pairs passed.
321 // See https://arduinojson.org/v6/assistant/ for more information on the needed size for the JsonDocument
322 size_t const size = Helper::distance(first, last);
323 TBJsonDocument jsonBuffer(JSON_OBJECT_SIZE(size));
324#else
325 StaticJsonDocument<JSON_OBJECT_SIZE(MaxFieldsAmount)> jsonBuffer;
326#endif // !THINGSBOARD_ENABLE_DYNAMIC
327
328 for (auto it = first; it != last; ++it) {
329 auto const & data = *it;
330 if (!data.SerializeKeyValue(jsonBuffer)) {
331 Logger::println(UNABLE_TO_SERIALIZE);
332 return false;
333 }
334 }
335
336 return telemetry ? sendTelemetryJson(jsonBuffer, Helper::Measure_Json(jsonBuffer)) : sendAttributeJson(jsonBuffer, Helper::Measure_Json(jsonBuffer));
337 }
338
345 template<typename T>
346 bool sendKeyValue(char const * const key, T value, bool telemetry = true) {
347 Telemetry const t(key, value);
348 if (t.IsEmpty()) {
349 // Message is ignored and not sent at all.
350 return false;
351 }
352
353 StaticJsonDocument<JSON_OBJECT_SIZE(1)> jsonBuffer;
354 if (!t.SerializeKeyValue(jsonBuffer)) {
355 Logger::printfln(UNABLE_TO_SERIALIZE);
356 return false;
357 }
358
359 return telemetry ? sendTelemetryJson(jsonBuffer, Helper::Measure_Json(jsonBuffer)) : sendAttributeJson(jsonBuffer, Helper::Measure_Json(jsonBuffer));
360 }
